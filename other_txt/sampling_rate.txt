# SQL query
select
    measurement_time,
    max(case when i.input_name='MOUSE(0)' then u.value else -1 end) as 'x',
    max(case when i.input_name='MOUSE(1)' then u.value else -1 end) as 'y'
from inputs i inner join counters_ull_time_data u on i.id_input=u.id_input
   group by measurement_time
   order by measurement_time
;
